{{/* Decide whether to apply a drop cap to a note. Criteria: ≥500 words (short notes look odd with a giant letter), doesn't start with a list, first character is uppercase, and the second character isn't an apostrophe (avoids wrapping the first letter of contractions like "'Twas"). */}} {{- $trimmed := .Content | strings.TrimLeft "\n\r\t " -}} {{- $startsWithList := or (hasPrefix $trimmed " content for character checks (skip headings etc.) */}} {{- $afterP := replaceRE `(?s)^.*?]*>` "" .Content -}} {{- $firstPText := replaceRE `(?s)

.*$` "" $afterP | plainify | replaceRE `’|‘` "'" | strings.TrimLeft "\n\r\t " -}} {{- $first := substr $firstPText 0 1 -}} {{- $second := substr $firstPText 1 1 -}} {{- $isUpper := and (eq $first (upper $first)) (ne $first (lower $first)) -}} {{- $hasApostrophe := eq $second "'" -}} {{- return (and (ge .WordCount 500) (not $startsWithList) $isUpper (not $hasApostrophe)) -}}
