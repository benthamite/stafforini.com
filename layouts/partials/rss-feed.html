{{/* Shared RSS feed structure. Both _default/rss.xml and quotes/rss.xml compute their page selection, then call this partial to render the full RSS document (channel metadata + items). Parameters (passed as a dict): .context - the Hugo page context (for title, permalink, description, etc.) .pages - the pre-filtered, pre-sorted, pre-limited slice of pages */}} {{- $ctx := .context -}} {{- $pages := .pages -}} {{- printf "" | safeHTML }} {{ if eq $ctx.Title $ctx.Site.Title }}{{ $ctx.Site.Title }}{{ else }}{{ with $ctx.Title }}{{ . }} · {{ end }}{{ $ctx.Site.Title }}{{ end }} {{ $ctx.Permalink }} {{ with $ctx.Description }}{{ . }}{{ else }}{{ with $ctx.Site.Params.description }}{{ . }}{{ end }}{{ end }} Hugo -- gohugo.io {{ $ctx.Site.LanguageCode }}{{ with $ctx.Site.Params.author }}{{ with .email }} {{ . }}{{ with $ctx.Site.Params.author.name }} ({{ . }}){{ end }}{{ end }}{{ with .email }} {{ . }}{{ with $ctx.Site.Params.author.name }} ({{ . }}){{ end }}{{ end }}{{ end }}{{ with $ctx.Site.Copyright }} {{ . }}{{ end }}{{ if not $ctx.Date.IsZero }} {{ $ctx.Date.Format "Mon, 02 Jan 2006 15:04:05 -0700" | safeHTML }}{{ end }} {{- with $ctx.OutputFormats.Get "RSS" }} {{ printf "" .Permalink .MediaType | safeHTML }} {{- end }} {{- range $pages }} {{- partial "rss-item.html" . }} {{- end }}
